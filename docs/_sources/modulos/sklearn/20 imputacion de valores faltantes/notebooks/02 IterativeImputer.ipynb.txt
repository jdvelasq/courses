    "    # The estimator to use at each step of the round-robin imputation\n",
    "    estimator=LinearRegression(),\n",
    "    # -------------------------------------------------------------------------\n",

    "    missing_values=np.nan,\n",
    "    # -------------------------------------------------------------------------\n",

    "    max_iter=10,\n",
    "    # -------------------------------------------------------------------------\n",
    "    # Number of other features to use to estimate the missing values of each\n",
